package main

  

import (

// --- Вендор: проверка статуса платежа по их API ---

  

type VendorStatus string

  

const (

    VendorPending VendorStatus = "PENDING"

    VendorPaid    VendorStatus = "PAID"

    VendorFailed  VendorStatus = "FAILED"

  

    byfferSize = 10

)

  

type VendorClient interface {

    GetTxStatus(ctx context.Context, vendorTxID string) (VendorStatus, error)

}

  

// --- Внутренняя очередь (обёртка над брокером) ---

  

type FulfillmentQueue interface {

    EnqueuePaymentPaid(ctx context.Context, msg PaymentPaidMessage) error

}

  

type PaymentPaidMessage struct {

    PaymentID string `json:"payment_id"`

    OrderID   string `json:"order_id"`

    Amount    int64  `json:"amount_cents"`

}

  

// --- Воркер опроса вендора ---

  

type Payment struct {

    ID          int64

    OrderID     string

    VendorTxID  string

    AmountCents int64

    Status      string

}

  

type VerificationWorker struct {

    db     *sql.DB

    vendor VendorClient

    queue  FulfillmentQueue

}

  

func NewVerificationWorker(db *sql.DB, v VendorClient, q FulfillmentQueue) *VerificationWorker {

    worker := VerificationWorker{db: db, vendor: v, queue: q)

    return &worker

}

  
  
  
  

// RunPollOnce обрабатывает порцию платежей, ожидающих ответа вендора.

func (w *VerificationWorker) RunPollOnce(ctx context.Context) error {

    rows, err := w.db.QueryContext(ctx, `

        SELECT id, order_id, vendor_tx_id, amount_cents, status

        FROM payments

        WHERE status = 'pending_vendor'

    `)

    if err != nil {

        return err

    }

    defer rows.Close()

  

    var batch []Payment

    for rows.Next() {

        var p Payment

        if err := rows.Scan(&p.ID, &p.OrderID, &p.VendorTxID, &p.AmountCents, &p.Status); err != nil {

            return err

        }

        batch = append(batch, p)

    }

    if err := rows.Err(); err != nil {

        return err

    }



    for _, p := range batch {
	    if err := w.processOne(ctx, p); err != nil {

                log.Printf("payment id=%d: %v", p.ID, err)

            }

    return nil

}

  
  

func (w *VerificationWorker) processOne()  {

    st, err := w.vendor.GetTxStatus(ctx, p.VendorTxID)

    if err != nil {

        return fmt.Errorf("vendor: %w", err)

    }

  

    if st == VendorFailed {

    _, e := w.db.ExecContext(ctx, `

                UPDATE payments SET status = 'failed', updated_at = NOW()

                WHERE id = $1 AND status = 'pending_vendor'

        `, p.ID) 

        return e

    }



    msg := PaymentPaidMessage{

        PaymentID: fmt.Sprintf("%d", p.ID),

        OrderID:   p.OrderID,

        Amount:    p.AmountCents,

    }

  

    if _, err := w.db.ExecContext(ctx, `

        UPDATE payments

        SET status = 'paid', paid_at = NOW(), updated_at = NOW()

        WHERE id = $1

    `, p.ID); err != nil {

        return fmt.Errorf("db mark paid: %w", err)

    }

  



  
  

    if err := w.queue.EnqueuePaymentPaid(ctx, msg); err != nil {

        return fmt.Errorf("queue: %w", err)

    }

  

    return nil

}
